use-- CREATE THE USER TABLE
CREATE TABLE USER (
    USER_ID VARCHAR(50) NOT NULL PRIMARY KEY,
    F_NAME VARCHAR(50),
    L_NAME VARCHAR(50),
    EMAIL VARCHAR(60),
    PASSWORD VARCHAR(70));


-- CREATE THE OWNER TABLE
CREATE TABLE OWNER (
    OWNER_ID VARCHAR(60) NOT NULL PRIMARY KEY,
    USER_ID VARCHAR(70),
    F_NAME VARCHAR(80),
    L_NAME VARCHAR(80),
    PH_NO BIGINT,
    FOREIGN KEY (USER_ID) REFERENCES USER(USER_ID));


-- CREATE THE HOUSE_DETAILS TABLE
CREATE TABLE HOUSE_DETAILS (
    HOUSE_ID VARCHAR(60) NOT NULL PRIMARY KEY,
    OWNER_ID VARCHAR(60) NOT NULL,
    HOUSE_TYPE VARCHAR(80),
    ADDRESS VARCHAR(80),
    IMAGE_DATA LONGTEXT,
    FOREIGN KEY (OWNER_ID) REFERENCES OWNER(OWNER_ID));


-- CREATE THE RENT TABLE
CREATE TABLE RENT (
    HOUSE_ID VARCHAR(60) NOT NULL,
    OWNER_ID VARCHAR(60) NOT NULL,
    R_PRICE BIGINT,
    ADVANCE_AMT BIGINT,
    PRIMARY KEY (HOUSE_ID, OWNER_ID),
    FOREIGN KEY (HOUSE_ID) REFERENCES HOUSE_DETAILS(HOUSE_ID),
    FOREIGN KEY (OWNER_ID) REFERENCES OWNER(OWNER_ID));


-- CREATE THE LEASE TABLE
CREATE TABLE LEASE (
    HOUSE_ID VARCHAR(60) NOT NULL,
    OWNER_ID VARCHAR(60) NOT NULL,
    L_PRICE BIGINT,
    PERIOD VARCHAR(50),
    PRIMARY KEY (HOUSE_ID, OWNER_ID),
    FOREIGN KEY (HOUSE_ID) REFERENCES HOUSE_DETAILS(HOUSE_ID),
    FOREIGN KEY (OWNER_ID) REFERENCES OWNER(OWNER_ID));


-- CREATE THE TENANT TABLE
CREATE TABLE TENANT (
    USER_ID VARCHAR(70) NOT NULL PRIMARY KEY,
    F_NAME VARCHAR(80),
    L_NAME VARCHAR(80),
    PH_NO BIGINT);
